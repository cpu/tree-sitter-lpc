// prolang.y `struct_name` (line 17264) accepts:
//   - bare `Name`
//   - `::Name` (next outer same-named struct)
//   - `scope::Name` where scope is `efun` or `sefun`
// in both type position (`struct sefun::Foo`) and struct-literal position
// (`(<sefun::Foo> ...)`).

struct sefun::SomeStruct g1;
struct ::NewStruct g2;

void f() {
  struct ::SomeStruct mine = (<::SomeStruct> 1, 2);
  struct sefun::SomeStruct s = (<sefun::SomeStruct> "Test");
  return s;
}
