// list optional — without it the inline closure synthesizes $1..$9 as
// implicit arguments (mixed). t-mantis.c and t-0000841/closure_crash.c
// stress the variants below.

closure a = function void { return $1; };
closure b = function : int x { return x; };
closure c = function void : int x {};
closure d = function : int i = 1; mixed j = 2 { return i + j; };
closure e = function int : int x = 5 { return x; };

void f() {
  return funcall(funcall(
    function :
      mixed i = 1;
      mixed j = function :
                  mixed j = function { return i; }
                { return j; };
      { return j; }
  ));
}
