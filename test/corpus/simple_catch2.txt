#define MAX_LOG_SIZE 50000

x() {
  string file_name;
  int *st;
  if (sizeof(st = get_dir(file_name,2) ) && st[0] > MAX_LOG_SIZE) {
    catch(rename(file_name, file_name + ".old")); /* No panic if failure */
  }
}
