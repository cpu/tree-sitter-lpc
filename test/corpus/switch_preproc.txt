#define TRACE_TYPE_SYMBOL 0
#define TRACE_TYPE_SEFUN  1
#define TRACE_TYPE_EFUN   2
#define TRACE_TYPE_LAMBDA 3
#define TRACE_TYPE_LFUN   4

string x(int typ) {
  switch(typ) {
    case TRACE_TYPE_SYMBOL: return "symbol";
    case TRACE_TYPE_SEFUN:  return "sefun";
    case TRACE_TYPE_EFUN:   return "efun";
    case TRACE_TYPE_LAMBDA: return "lambda";
    case TRACE_TYPE_LFUN:   return "lfun";
    default:                return "unknown";
  }
}
